Like the format! macro, but for creating bumpalo::collections::Strings.

use bumpalo::Bump;

let b = Bump::new();

let who = "World";
let s = bumpalo::format!(in &b, "Hello, {}!", who);
assert_eq!(s, "Hello, World!")
