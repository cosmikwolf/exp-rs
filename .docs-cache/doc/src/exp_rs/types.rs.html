73/// Abstract Syntax Tree (AST) node representing an expression.
74///
75/// The AST is the core data structure used for representing parsed expressions.
76/// Each variant of this enum represents a different type of expression node,
77/// forming a tree structure that can be evaluated to produce a result.
78///
79/// This type uses arena allocation for all strings and recursive structures,
80/// eliminating all dynamic allocations during evaluation.
81///
82/// Using repr(C) with explicit discriminant type and alignment to avoid ARM alignment issues
83#[derive(Debug)]
84#[repr(C, align(8))]
85pub enum AstExpr<'arena> {
86    /// A literal numerical value.
87    ///
88    /// Examples: `3.14`, `42`, `-1.5`
89    Constant(Real),
90
91    /// A named variable reference.
92    ///
93    /// Examples: `x`, `temperature`, `result`
94    Variable(&'arena str),
95
96    /// A function call with a name and list of argument expressions.
97    ///
98    /// Examples: `sin(x)`, `max(a, b)`, `sqrt(x*x + y*y)`
99    Function {
100        /// The name of the function being called
101        name: &'arena str,
102        /// The arguments passed to the function
103        args: &'arena [AstExpr<'arena>],
104    },
105
106    /// An array element access.
107    ///
108    /// Examples: `array[0]`, `values[i+1]`
109    Array {
110        /// The name of the array
111        name: &'arena str,
112        /// The expression for the index
113        index: &'arena AstExpr<'arena>,
114    },
115
116    /// An attribute access on an object.
117    ///
118    /// Examples: `point.x`, `settings.value`
119    Attribute {
120        /// The base object name
121        base: &'arena str,
122        /// The attribute name
123        attr: &'arena str,
124    },
125
126    /// A logical operation with short-circuit evaluation.
127    ///
128    /// Represents logical AND (`&&`) and OR (`||`) operations with short-circuit behavior.
129    /// Unlike function-based operators, these operators have special evaluation semantics
130    /// where the right operand may not be evaluated based on the value of the left operand.
131    ///
132    /// # Examples
133    ///
134    /// - `a && b`: Evaluates `a`, then evaluates `b` only if `a` is non-zero (true)
135    /// - `c || d`: Evaluates `c`, then evaluates `d` only if `c` is zero (false)
136    /// - `x > 0 && y < 10`: Checks if both conditions are true, with short-circuit
137    /// - `flag || calculate_value()`: Skips calculation if flag is true
138    ///
139    /// # Boolean Logic
140    ///
141    /// The engine represents boolean values as floating-point numbers:
142    /// - `0.0` is considered `false`
143    /// - Any non-zero value is considered `true`, typically `1.0` is used
144    ///
145    /// # Operator Precedence
146    ///
147    /// `&&` has higher precedence than `||`, consistent with most programming languages:
148    /// - `a || b && c` is interpreted as `a || (b && c)`
149    /// - Use parentheses to override default precedence: `(a || b) && c`
150    LogicalOp {
151        /// The logical operator (AND or OR)
152        op: LogicalOperator,
153        /// The left operand (always evaluated)
154        left: &'arena AstExpr<'arena>,
155        /// The right operand (conditionally evaluated based on left value)
156        right: &'arena AstExpr<'arena>,
157    },
158
159    /// A ternary conditional operation (condition ? true_expr : false_expr).
160    ///
161    /// Represents a conditional expression with three parts: a condition to evaluate,
162    /// an expression to return if the condition is true, and an expression to return
163    /// if the condition is false. This uses short-circuit evaluation, meaning only
164    /// the relevant branch is evaluated.
165    ///
166    /// # Examples
167    ///
168    /// - `x > 0 ? 1 : -1`: Returns 1 if x is positive, -1 otherwise
169    /// - `flag ? value1 : value2`: Chooses between two values based on flag
170    /// - `a > b ? a : b`: Returns the maximum of a and b
171    ///
172    /// # Boolean Logic
173    ///
174    /// Like logical operations, the ternary operator uses floating-point values for boolean logic:
175    /// - `0.0` is considered `false`
176    /// - Any non-zero value is considered `true`
177    ///
178    /// # Short-Circuit Evaluation
179    ///
180    /// Only one branch is evaluated based on the condition result:
181    /// - If condition is non-zero (true), only the true_branch is evaluated
182    /// - If condition is zero (false), only the false_branch is evaluated
183    ///
184    /// # Operator Precedence
185    ///
186    /// The ternary operator has low precedence:
187    /// - `a + b ? c : d * e` is interpreted as `(a + b) ? c : (d * e)`
188    /// - Use parentheses for clarity when nesting operations
189    Conditional {
190        /// The condition expression to evaluate
191        condition: &'arena AstExpr<'arena>,
192        /// Expression to evaluate if condition is true (non-zero)
193        true_branch: &'arena AstExpr<'arena>,
194        /// Expression to evaluate if condition is false (zero)
195        false_branch: &'arena AstExpr<'arena>,
196    },
197}

236impl<'arena> AstExpr<'arena> {
237    /// Helper method that raises a constant expression to a power.
238    ///
239    /// This is primarily used in testing to evaluate power operations on constants.
240    /// For non-constant expressions, it returns 0.0 as a default value.
241    ///
242    /// # Parameters
243    ///
244    /// * `exp` - The exponent to raise the constant to
245    ///
246    /// # Returns
247    ///
248    /// The constant raised to the given power, or 0.0 for non-constant expressions
249    pub fn pow(&self, exp: Real) -> Real {
250        match self {
251            AstExpr::Constant(val) => {
252                #[cfg(all(feature = "libm", feature = "f32"))]
253                {
254                    libm::powf(*val, *exp)
255                }
256                #[cfg(all(feature = "libm", not(feature = "f32")))]
257                {
258                    libm::pow(*val, exp)
259                }
260                #[cfg(all(not(feature = "libm"), test))]
261                {
262                    val.powf(*exp)
263                } // Use std::powf when in test mode
264                #[cfg(all(not(feature = "libm"), not(test)))]
265                {
266                    // Without libm and not in tests, limited power implementation
267                    if exp == 0.0 {
268                        1.0
269                    } else if exp == 1.0 {
270                        val
271                    } else if exp == 2.0 {
272                        val * val
273                    } else {
274                        0.0
275                    } // This functionality requires explicit registration
276                }
277            }
278            _ => 0.0, // Default for non-constant expressions
279        }
280    }
281}

447/// Defines the type of logical operation.
448///
449/// Used by the `LogicalOp` variant of `AstExpr` to specify which logical operation
450/// should be performed with short-circuit evaluation semantics.
451///
452/// # Short-Circuit Evaluation
453///
454/// Short-circuit evaluation is an optimization technique where the second operand
455/// of a logical operation is evaluated only when necessary:
456///
457/// - For `&&` (AND): If the left operand is false, the result is false regardless
458///   of the right operand, so the right operand is not evaluated.
459///
460/// - For `||` (OR): If the left operand is true, the result is true regardless
461///   of the right operand, so the right operand is not evaluated.
462///
463/// This behavior is particularly useful for:
464///
465/// 1. Performance optimization - avoid unnecessary calculation
466/// 2. Conditional execution - control evaluation of expressions
467/// 3. Safe guards - prevent errors (e.g., division by zero)
468///
469/// # Boolean Representation
470///
471/// In this expression engine, boolean values are represented as floating-point numbers:
472///
473/// - `0.0` represents `false`
474/// - Any non-zero value (typically `1.0`) represents `true`
475#[derive(Clone, Debug, PartialEq)]
476pub enum LogicalOperator {
477    /// Logical AND (&&) - evaluates to true only if both operands are true.
478    /// Short-circuits if the left operand is false.
479    ///
480    /// Examples:
481    /// - `1 && 1` evaluates to `1.0` (true)
482    /// - `1 && 0` evaluates to `0.0` (false)
483    /// - `0 && expr` evaluates to `0.0` without evaluating `expr`
484    And,
485
486    /// Logical OR (||) - evaluates to true if either operand is true.
487    /// Short-circuits if the left operand is true.
488    ///
489    /// Examples:
490    /// - `1 || 0` evaluates to `1.0` (true)
491    /// - `0 || 0` evaluates to `0.0` (false)
492    /// - `1 || expr` evaluates to `1.0` without evaluating `expr`
493    Or,
494}
495
496/// Implements Display for LogicalOperator to use in error messages.
497impl core::fmt::Display for LogicalOperator {
498    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
499        match self {
500            LogicalOperator::And => write!(f, "&&"),
501            LogicalOperator::Or => write!(f, "||"),
502        }
503    }
504}

506/// Represents a native Rust function that can be registered with the evaluation context.
507///
508/// Native functions allow users to extend the expression evaluator with custom
509/// functionality written in Rust. These functions can be called from within expressions
510/// like any built-in function.
511///
512/// # Example
513///
514/// ```
515/// # use exp_rs::{EvalContext, Real};
516/// # use exp_rs::engine::interp;
517/// # use std::rc::Rc;
518/// let mut ctx = EvalContext::new();
519///
520/// // Register a custom function that calculates the hypotenuse
521/// ctx.register_native_function(
522///     "hypotenuse",     // Function name
523///     2,                // Takes 2 arguments
524///     |args: &[Real]| { // Implementation
525///         (args[0] * args[0] + args[1] * args[1]).sqrt()
526///     }
527/// );
528///
529/// // Use the function in an expression
530/// let result = interp("hypotenuse(3, 4)", Some(Rc::new(ctx))).unwrap();
531/// assert_eq!(result, 5.0);
532/// ```
533#[derive(Clone)]
534pub struct NativeFunction {
535    /// Number of arguments the function takes.
536    pub arity: usize,
537
538    /// The actual implementation of the function as a Rust closure.
539    pub implementation: Rc<dyn Fn(&[Real]) -> Real>,
540
541    /// The name of the function as it will be used in expressions.
542    pub name: FunctionName,
543
544    /// Optional description of what the function does.
545    pub description: Option<String>,
546}

554/// Represents a function defined by an expression string rather than Rust code.
555///
556/// Expression functions allow users to define custom functions using the expression
557/// language itself. These functions are compiled once when registered and can be called
558/// from other expressions. They support parameters and can access variables from the
559/// evaluation context.
560///
561/// # Example
562///
563/// ```
564/// # use exp_rs::{EvalContext, Real};
565/// # use exp_rs::engine::interp;
566/// # use std::rc::Rc;
567/// let mut ctx = EvalContext::new();
568///
569/// // Register a function to calculate the area of a circle
570/// ctx.register_expression_function(
571///     "circle_area",            // Function name
572///     &["radius"],              // Parameter names
573///     "pi * radius * radius"    // Function body as an expression
574/// ).unwrap();
575///
576/// // Use the function in another expression
577/// let result = interp("circle_area(2)", Some(Rc::new(ctx))).unwrap();
578/// assert!(result > 12.56 && result < 12.57); // π * 4 ≈ 12.566
579/// ```
580#[derive(Clone)]
581pub struct ExpressionFunction {
582    /// The name of the function as it will be used in expressions.
583    pub name: FunctionName,
584
585    /// The parameter names that the function accepts.
586    pub params: Vec<String>,
587
588    /// The original expression string defining the function body.
589    pub expression: String,
590
591    /// Optional description of what the function does.
592    pub description: Option<String>,
593}

595/// Internal representation of a variable in the evaluation system.
596///
597/// This is an implementation detail and should not be used directly by library users.
598/// Variables are normally managed through the `EvalContext` interface.
599#[doc(hidden)]
600pub struct Variable<'a> {

602    pub name: Cow<'a, str>,
603
604    /// Internal address/identifier for the variable.

607    /// Function associated with the variable (if any).
608    pub function: fn(Real, Real) -> Real,
609
610    /// Context or associated AST nodes.
611    pub context: Vec<AstExpr<'a>>,
612}
