26pub fn eval_variable(
27    name: &str,
28    ctx: Option<Rc<EvalContext>>,
29    var_cache: &mut BTreeMap<String, Real>,
30) -> Result<Real, ExprError> {
31    // Handle built-in constants first (pi, e) - these don't need context
32    if name == "pi" {
33        #[cfg(feature = "f32")]
34        return Ok(core::f32::consts::PI);
35        #[cfg(not(feature = "f32"))]
36        return Ok(core::f64::consts::PI);
37    } else if name == "e" {
38        #[cfg(feature = "f32")]
39        return Ok(core::f32::consts::E);
40        #[cfg(not(feature = "f32"))]
41        return Ok(core::f64::consts::E);
42    }
43
44    // Check local cache first
45    if let Some(val) = var_cache.get(name).copied() {
46        return Ok(val);
47    }
48
49    // Use context helper methods for variable/constant lookup
50    if let Some(ctx_ref) = ctx.as_deref() {
51        // For expression functions, we need to check the immediate variables first
52        // before checking parent contexts to avoid shadowing issues
53        if let Ok(key) = name.try_into_heapless() {
54            if let Some(val) = ctx_ref.variables.get(&key) {
55                // Found in immediate context, return it
56                var_cache.insert(name.to_string(), *val);
57                return Ok(*val);
58            }
59        }
60
61        // Then check constants in the immediate context
62        if let Ok(key) = name.try_into_heapless() {
63            if let Some(val) = ctx_ref.constants.get(&key) {
64                var_cache.insert(name.to_string(), *val);
65                return Ok(*val);
66            }
67        }
68
69        // Now check the parent chain using the helper methods
70        // which will properly handle the parent chain
71        if let Some(val) = ctx_ref.get_variable(name) {
72            var_cache.insert(name.to_string(), val);
73            return Ok(val);
74        } else if let Some(val) = ctx_ref.get_constant(name) {
75            var_cache.insert(name.to_string(), val);
76            return Ok(val);
77        }
78    }
79
80    // If not a constant and not found in context, check if it looks like a function name
81    let is_potential_function_name = match name {
82        "sin" | "cos" | "tan" | "asin" | "acos" | "atan" | "atan2" | "sinh" | "cosh" | "tanh"
83        | "exp" | "log" | "log10" | "ln" | "sqrt" | "abs" | "ceil" | "floor" | "pow" | "neg"
84        | "," | "comma" | "+" | "-" | "*" | "/" | "%" | "^" | "max" | "min" => true,
85        _ => false,
86    };
87
88    if is_potential_function_name && name.len() > 1 {
89        return Err(ExprError::Syntax(format!(
90            "Function '{}' used without arguments",
91            name
92        )));
93    }
94
95    Err(ExprError::UnknownVariable {
96        name: name.to_string(),
97    })
98}

100pub fn eval_function<'a>(
101    name: &str,
102    args: &[AstExpr],
103    ctx: Option<Rc<EvalContext>>,
104    func_cache: &mut BTreeMap<String, Option<FunctionCacheEntry>>,
105    var_cache: &mut BTreeMap<String, Real>,
106) -> Result<Real, ExprError> {
107    // Try to get cached function info first
108    if let Some(entry) = func_cache.get(name) {
109        if let Some(cached_fn) = entry {
110            return match cached_fn.clone() {
111                FunctionCacheEntry::User(user_fn) => {
112                    eval_custom_function(name, args, ctx.clone(), func_cache, var_cache, &user_fn)
113                }
114                FunctionCacheEntry::Expression(expr_fn) => {
115                    eval_custom_function(name, args, ctx.clone(), func_cache, var_cache, &expr_fn)
116                }
117                FunctionCacheEntry::Native(native_fn) => {
118                    eval_native_function(name, args, ctx.clone(), func_cache, var_cache, &native_fn)
119                }
120            };
121        }
122    }
123
124    // If not in cache, look up in context and create cache entry
125    let entry = if let Some(ctx_ref) = ctx.as_ref() {
126        // Create fully owned entries from the context's data
127        if let Some(expr_fn) = ctx_ref.get_expression_function(name) {
128            // ExpressionFunction is already Clone so we can use it directly
129            Some(FunctionCacheEntry::Expression(expr_fn.clone()))
130        } else if let Some(native_fn) = ctx_ref.get_native_function(name) {
131            // Convert to our owned version
132            let owned_fn = OwnedNativeFunction::from(native_fn);
133            Some(FunctionCacheEntry::Native(owned_fn))
134        } else { ctx_ref.get_user_function(name).map(|user_fn| FunctionCacheEntry::User(user_fn.clone())) }
135    } else {
136        None
137    };
138
139    // Cache the entry without borrowing ctx
140    func_cache.insert(name.to_string(), entry.clone());
141
142    // Process the entry
143    if let Some(func_entry) = entry {
144        match func_entry {
145            FunctionCacheEntry::User(user_fn) => {
146                eval_custom_function(name, args, ctx.clone(), func_cache, var_cache, &user_fn)
147            }
148            FunctionCacheEntry::Expression(expr_fn) => {
149                eval_custom_function(name, args, ctx.clone(), func_cache, var_cache, &expr_fn)
150            }
151            FunctionCacheEntry::Native(native_fn) => {
152                eval_native_function(name, args, ctx.clone(), func_cache, var_cache, &native_fn)
153            }
154        }
155    } else {
156        // Rest of the function remains the same...
157        // Fallback to built-in functions...
158        #[cfg(feature = "libm")]
159        {
160            // First check known functions against their expected arity
161            // This allows us to return InvalidFunctionCall errors rather than UnknownFunction
162            let single_arg_funcs = [
163                "sin", "cos", "tan", "asin", "acos", "atan", "sinh", "cosh", "tanh", "exp", "log",
164                "ln", "log10", "sqrt", "abs", "ceil", "floor", "neg",
165            ];
166
167            let two_arg_funcs = [
168                "+", "-", "*", "/", "^", "pow", "max", "min", "%", ",", "comma", "atan2",
169            ];
170
171            // Check single-arg functions with wrong arity
172            if single_arg_funcs.contains(&name) && args.len() != 1 {
173                return Err(ExprError::InvalidFunctionCall {
174                    name: name.to_string(),
175                    expected: 1,
176                    found: args.len(),
177                });
178            }
179
180            // Check two-arg functions with wrong arity
181            if two_arg_funcs.contains(&name) && args.len() != 2 {
182                return Err(ExprError::InvalidFunctionCall {
183                    name: name.to_string(),
184                    expected: 2,
185                    found: args.len(),
186                });
187            }
188
189            // Now evaluate with correct arity
190            if args.len() == 1 {
191                // Single-arg built-in functions
192                let arg_val = eval_ast_inner(&args[0], ctx.clone(), func_cache, var_cache)?;
193                match name {
194                    "sin" => {
195                        #[cfg(feature = "f32")]
196                        {
197                            return Ok(libm::sinf(arg_val));
198                        }
199                        #[cfg(not(feature = "f32"))]
200                        {
201                            return Ok(libm::sin(arg_val));
202                        }
203                    }
204                    "cos" => {
205                        #[cfg(feature = "f32")]
206                        {
207                            return Ok(libm::cosf(arg_val));
208                        }
209                        #[cfg(not(feature = "f32"))]
210                        {
211                            return Ok(libm::cos(arg_val));
212                        }
213                    }
214                    "tan" => {
215                        #[cfg(feature = "f32")]
216                        {
217                            return Ok(libm::tanf(arg_val));
218                        }
219                        #[cfg(not(feature = "f32"))]
220                        {
221                            return Ok(libm::tan(arg_val));
222                        }
223                    }
224                    "asin" => {
225                        #[cfg(feature = "f32")]
226                        {
227                            return Ok(libm::asinf(arg_val));
228                        }
229                        #[cfg(not(feature = "f32"))]
230                        {
231                            return Ok(libm::asin(arg_val));
232                        }
233                    }
234                    "acos" => {
235                        #[cfg(feature = "f32")]
236                        {
237                            return Ok(libm::acosf(arg_val));
238                        }
239                        #[cfg(not(feature = "f32"))]
240                        {
241                            return Ok(libm::acos(arg_val));
242                        }
243                    }
244                    "atan" => {
245                        #[cfg(feature = "f32")]
246                        {
247                            return Ok(libm::atanf(arg_val));
248                        }
249                        #[cfg(not(feature = "f32"))]
250                        {
251                            return Ok(libm::atan(arg_val));
252                        }
253                    }
254                    "sinh" => {
255                        #[cfg(feature = "f32")]
256                        {
257                            return Ok(libm::sinhf(arg_val));
258                        }
259                        #[cfg(not(feature = "f32"))]
260                        {
261                            return Ok(libm::sinh(arg_val));
262                        }
263                    }
264                    "cosh" => {
265                        #[cfg(feature = "f32")]
266                        {
267                            return Ok(libm::coshf(arg_val));
268                        }
269                        #[cfg(not(feature = "f32"))]
270                        {
271                            return Ok(libm::cosh(arg_val));
272                        }
273                    }
274                    "tanh" => {
275                        #[cfg(feature = "f32")]
276                        {
277                            return Ok(libm::tanhf(arg_val));
278                        }
279                        #[cfg(not(feature = "f32"))]
280                        {
281                            return Ok(libm::tanh(arg_val));
282                        }
283                    }
284                    "exp" => {
285                        #[cfg(feature = "f32")]
286                        {
287                            return Ok(libm::expf(arg_val));
288                        }
289                        #[cfg(not(feature = "f32"))]
290                        {
291                            return Ok(libm::exp(arg_val));
292                        }
293                    }
294                    "log" | "ln" => {
295                        #[cfg(feature = "f32")]
296                        {
297                            return Ok(libm::logf(arg_val));
298                        }
299                        #[cfg(not(feature = "f32"))]
300                        {
301                            return Ok(libm::log(arg_val));
302                        }
303                    }
304                    "log10" => {
305                        #[cfg(feature = "f32")]
306                        {
307                            return Ok(libm::log10f(arg_val));
308                        }
309                        #[cfg(not(feature = "f32"))]
310                        {
311                            return Ok(libm::log10(arg_val));
312                        }
313                    }
314                    "sqrt" => {
315                        #[cfg(feature = "f32")]
316                        {
317                            return Ok(libm::sqrtf(arg_val));
318                        }
319                        #[cfg(not(feature = "f32"))]
320                        {
321                            return Ok(libm::sqrt(arg_val));
322                        }
323                    }
324                    "abs" => return Ok(arg_val.abs()),
325                    "ceil" => {
326                        #[cfg(feature = "f32")]
327                        {
328                            return Ok(libm::ceilf(arg_val));
329                        }
330                        #[cfg(not(feature = "f32"))]
331                        {
332                            return Ok(libm::ceil(arg_val));
333                        }
334                    }
335                    "floor" => {
336                        #[cfg(feature = "f32")]
337                        {
338                            return Ok(libm::floorf(arg_val));
339                        }
340                        #[cfg(not(feature = "f32"))]
341                        {
342                            return Ok(libm::floor(arg_val));
343                        }
344                    }
345                    "neg" => return Ok(-arg_val),
346                    _ => {}
347                }
348            } else if args.len() == 2 {
349                // Two-arg built-in functions
350                let mut arg_vals = [0.0; 2];
351                arg_vals[0] = eval_ast_inner(&args[0], ctx.clone(), func_cache, var_cache)?;
352                arg_vals[1] = eval_ast_inner(&args[1], ctx.clone(), func_cache, var_cache)?;
353                match name {
354                    // Basic arithmetic operators
355                    "+" => return Ok(arg_vals[0] + arg_vals[1]),
356                    "-" => return Ok(arg_vals[0] - arg_vals[1]),
357                    "*" => return Ok(arg_vals[0] * arg_vals[1]),
358                    "/" => {
359                        if arg_vals[1] == 0.0 {
360                            return Err(ExprError::DivideByZero);
361                        }
362                        return Ok(arg_vals[0] / arg_vals[1]);
363                    },
364                    "^" | "pow" => {
365                        #[cfg(feature = "f32")]
366                        {
367                            return Ok(libm::powf(arg_vals[0], arg_vals[1]));
368                        }
369                        #[cfg(not(feature = "f32"))]
370                        {
371                            return Ok(libm::pow(arg_vals[0], arg_vals[1]));
372                        }
373                    }
374                    "max" => return Ok(arg_vals[0].max(arg_vals[1])),
375                    "min" => return Ok(arg_vals[0].min(arg_vals[1])),
376                    "%" => {
377                        if arg_vals[1] == 0.0 {
378                            return Err(ExprError::DivideByZero);
379                        }
380                        return Ok(arg_vals[0] % arg_vals[1]);
381                    },
382                    "," | "comma" => return Ok(arg_vals[1]),
383
384                    // Comparison operators - return 1.0 for true and 0.0 for false
385                    "<" => return Ok(if arg_vals[0] < arg_vals[1] { 1.0 } else { 0.0 }),
386                    ">" => return Ok(if arg_vals[0] > arg_vals[1] { 1.0 } else { 0.0 }),
387                    "<=" => return Ok(if arg_vals[0] <= arg_vals[1] { 1.0 } else { 0.0 }),
388                    ">=" => return Ok(if arg_vals[0] >= arg_vals[1] { 1.0 } else { 0.0 }),
389                    "==" => return Ok(if arg_vals[0] == arg_vals[1] { 1.0 } else { 0.0 }),
390                    "!=" | "<>" => return Ok(if arg_vals[0] != arg_vals[1] { 1.0 } else { 0.0 }),
391
392                    "atan2" => {
393                        #[cfg(feature = "f32")]
394                        {
395                            return Ok(libm::atan2f(arg_vals[0], arg_vals[1]));
396                        }
397                        #[cfg(not(feature = "f32"))]
398                        {
399                            return Ok(libm::atan2(arg_vals[0], arg_vals[1]));
400                        }
401                    }
402                    _ => {}
403                }
404            }
405        }
406
407        // If we get here, the function is unknown
408        Err(ExprError::UnknownFunction {
409            name: name.to_string(),
410        })
411    }
412}

414pub fn eval_native_function<'a>(
415    name: &str,
416    args: &[AstExpr],
417    ctx: Option<Rc<EvalContext>>,
418    func_cache: &mut BTreeMap<String, Option<FunctionCacheEntry>>,
419    var_cache: &mut BTreeMap<String, Real>,
420    native_fn: &OwnedNativeFunction,
421) -> Result<Real, ExprError> {
422    if args.len() != native_fn.arity {
423        return Err(ExprError::InvalidFunctionCall {
424            name: name.to_string(),
425            expected: native_fn.arity,
426            found: args.len(),
427        });
428    }
429    let mut arg_values = Vec::with_capacity(args.len());
430    for arg in args.iter() {
431        let arg_val = eval_ast_inner(arg, ctx.clone(), func_cache, var_cache)?;
432        arg_values.push(arg_val);
433    }
434    Ok((native_fn.implementation)(&arg_values))
435}

437pub fn eval_array<'a>(
438    name: &str,
439    index: &AstExpr,
440    ctx: Option<Rc<EvalContext>>,
441    func_cache: &mut BTreeMap<String, Option<FunctionCacheEntry>>,
442    var_cache: &mut BTreeMap<String, Real>,
443) -> Result<Real, ExprError> {
444    let idx = eval_ast_inner(index, ctx.clone(), func_cache, var_cache)? as usize;
445
446    if let Some(ctx_ref) = ctx.as_ref() {
447        if let Some(arr) = ctx_ref.get_array(name) {
448            if idx < arr.len() {
449                return Ok(arr[idx]);
450            } else {
451                return Err(ExprError::ArrayIndexOutOfBounds {
452                    name: name.to_string(),
453                    index: idx,
454                    len: arr.len(),
455                });
456            }
457        }
458    }
459    Err(ExprError::UnknownVariable {
460        name: name.to_string(),
461    })
462}

464pub fn eval_attribute(
465    base: &str,
466    attr: &str,
467    ctx: Option<Rc<EvalContext>>,
468) -> Result<Real, ExprError> {
469    if let Some(ctx_ref) = ctx.as_ref() {
470        if let Some(attr_map) = ctx_ref.get_attribute_map(base) {
471            if let Ok(attr_key) = attr.try_into_heapless() {
472                if let Some(val) = attr_map.get(&attr_key) {
473                    return Ok(*val);
474                }
475            }
476        }
477    }
478    Err(ExprError::AttributeNotFound {
479        base: base.to_string(),
480        attr: attr.to_string(),
481    })
