1//! Iterative AST evaluator
2//!
3//! This module implements an iterative (non-recursive) AST evaluator using
4//! an explicit stack. This approach eliminates stack overflow issues and
5//! provides better performance for deeply nested expressions.
6

24/// Maximum depth of the operation stack (prevents runaway evaluation)
25const MAX_STACK_DEPTH: usize = 1000;
26
27/// Initial capacity for stacks (tuned for typical expressions)
28const INITIAL_OP_CAPACITY: usize = 32;
29const INITIAL_VALUE_CAPACITY: usize = 16;
30
31/// Main iterative evaluation function
32pub fn eval_iterative<'arena>(
33    ast: &'arena AstExpr<'arena>,
34    ctx: Option<Rc<EvalContext>>,
35) -> Result<Real, ExprError> {
36    let mut engine = EvalEngine::new();
37    engine.eval(ast, ctx)
38}

40/// Reusable evaluation engine to avoid allocations
41pub struct EvalEngine<'arena> {
42    /// Operation stack
43    op_stack: Vec<EvalOp<'arena>>,
44    /// Value stack for intermediate results
45    value_stack: Vec<Real>,
46    /// Context management
47    ctx_stack: ContextStack,
48    /// Function cache
49    func_cache: BTreeMap<HString, Option<FunctionCacheEntry>>,
50    /// Parameter overrides for batch evaluation (avoids context modification)
51    param_overrides: Option<FnvIndexMap<HString, Real, 16>>,
52    /// Optional arena for parsing expression functions on-demand
53    arena: Option<&'arena bumpalo::Bump>,
54    /// Cache for parsed expression functions
55    expr_func_cache: BTreeMap<HString, &'arena AstExpr<'arena>>,
56}
57
58impl<'arena> Default for EvalEngine<'arena> {
59    fn default() -> Self {
60        Self::new()
61    }
62}

91    /// Evaluate an expression
92    pub fn eval(&mut self, ast: &'arena AstExpr<'arena>, ctx: Option<Rc<EvalContext>>) -> Result<Real, ExprError> {
93        // Clear stacks but keep capacity
94        self.op_stack.clear();
95        self.value_stack.clear();
96        self.ctx_stack.clear();
97        self.func_cache.clear();
98        
99        // Initialize with root context
100        let root_ctx_id = self.ctx_stack.push_context(ctx)?;
101        
102        // Push initial operation (no clone needed - just use reference!)
103        self.op_stack.push(EvalOp::Eval { 
104            expr: ast, 
105            ctx_id: root_ctx_id 
106        });
107        
108        // Main evaluation loop
109        while let Some(op) = self.op_stack.pop() {
110            // Check depth limit
111            if self.op_stack.len() > MAX_STACK_DEPTH {
112                return Err(ExprError::RecursionLimit(
113                    format!("Maximum evaluation depth {} exceeded", MAX_STACK_DEPTH)
114                ));
115            }
116            
117            self.process_operation(op)?;
118        }
119        
120        // Result should be on top of value stack
121        self.value_stack.pop()
122            .ok_or_else(|| ExprError::Other("No result on value stack".to_string()))
123    }

125    /// Process a single operation
126    fn process_operation(&mut self, op: EvalOp<'arena>) -> Result<(), ExprError> {
127        match op {
128            EvalOp::Eval { expr, ctx_id } => {
129                self.process_eval(expr, ctx_id)?;
130            }
131            
132            EvalOp::ApplyUnary { op } => {
133                let operand = self.pop_value()?;
134                self.value_stack.push(op.apply(operand));
135            }
136            
137            EvalOp::CompleteBinary { op } => {
138                let right = self.pop_value()?;
139                let left = self.pop_value()?;
140                self.value_stack.push(op.apply(left, right));
141            }
142            
143            EvalOp::ShortCircuitAnd { right_expr, ctx_id } => {
144                let left_val = self.pop_value()?;
145                if left_val == 0.0 {
146                    // Short circuit - left is false, result is false
147                    self.value_stack.push(0.0);
148                } else {
149                    // Need to evaluate right side
150                    self.op_stack.push(EvalOp::CompleteAnd);
151                    self.op_stack.push(EvalOp::Eval { expr: right_expr, ctx_id });
152                }
153            }
154            
155            EvalOp::ShortCircuitOr { right_expr, ctx_id } => {
156                let left_val = self.pop_value()?;
157                if left_val != 0.0 {
158                    // Short circuit - left is true, result is true
159                    self.value_stack.push(1.0);
160                } else {
161                    // Need to evaluate right side
162                    self.op_stack.push(EvalOp::CompleteOr);
163                    self.op_stack.push(EvalOp::Eval { expr: right_expr, ctx_id });
164                }
165            }
166            
167            EvalOp::CompleteAnd => {
168                let right = self.pop_value()?;
169                let left = 1.0; // We know left was true or we wouldn't be here
170                self.value_stack.push(if left != 0.0 && right != 0.0 { 1.0 } else { 0.0 });
171            }
172            
173            EvalOp::CompleteOr => {
174                let right = self.pop_value()?;
175                let left = 0.0; // We know left was false or we wouldn't be here
176                self.value_stack.push(if left != 0.0 || right != 0.0 { 1.0 } else { 0.0 });
177            }
178            
179            EvalOp::LookupVariable { name, ctx_id } => {
180                self.process_variable_lookup(name, ctx_id)?;
181            }
182            
183            EvalOp::TernaryCondition { true_branch, false_branch, ctx_id } => {
184                let condition = self.pop_value()?;
185                if condition != 0.0 {
186                    self.op_stack.push(EvalOp::Eval { expr: true_branch, ctx_id });
187                } else {
188                    self.op_stack.push(EvalOp::Eval { expr: false_branch, ctx_id });
189                }
190            }
191            
192            EvalOp::AccessArray { array_name, ctx_id } => {
193                let index = self.pop_value()?;
194                self.process_array_access(array_name, index, ctx_id)?;
195            }
196            
197            EvalOp::AccessAttribute { object_name, attr_name, ctx_id } => {
198                self.process_attribute_access(object_name, attr_name, ctx_id)?;
199            }
200            
201            EvalOp::CollectFunctionArgs { name, total_args, mut args_so_far, ctx_id } => {
202                // Pop one argument from value stack and insert at the beginning
203                // to preserve the original order (since we evaluate args in reverse)
204                let arg = self.pop_value()?;
205                args_so_far.insert(0, arg);
206                
207                if args_so_far.len() == total_args {
208                    // All arguments collected, apply function
209                    self.op_stack.push(EvalOp::ApplyFunction {
210                        name,
211                        args_needed: total_args,
212                        args_collected: args_so_far,
213                        ctx_id,
214                    });
215                } else {
216                    // Still need more arguments
217                    self.op_stack.push(EvalOp::CollectFunctionArgs {
218                        name,
219                        total_args,
220                        args_so_far,
221                        ctx_id,
222                    });
223                }
224            }
225            
226            EvalOp::ApplyFunction { name, args_needed, args_collected, ctx_id } => {
227                self.process_function_call(name, args_needed, args_collected, ctx_id)?;
228            }
229        }
230        
231        Ok(())
232    }

234    /// Process an Eval operation by converting AST to stack operations
235    fn process_eval(&mut self, expr: &'arena AstExpr<'arena>, ctx_id: usize) -> Result<(), ExprError> {
236        match expr {
237            AstExpr::Constant(val) => {
238                self.value_stack.push(*val);
239            }
240            
241            AstExpr::Variable(name) => {
242                let hname = name.try_into_heapless()?;
243                self.op_stack.push(EvalOp::LookupVariable { 
244                    name: hname, 
245                    ctx_id 
246                });
247            }
248            
249            AstExpr::LogicalOp { op, left, right } => {
250                // Handle short-circuit operators
251                use crate::types::LogicalOperator;
252                match op {
253                    LogicalOperator::And => {
254                        self.op_stack.push(EvalOp::ShortCircuitAnd { 
255                            right_expr: right, 
256                            ctx_id 
257                        });
258                        self.op_stack.push(EvalOp::Eval { expr: left, ctx_id });
259                    }
260                    LogicalOperator::Or => {
261                        self.op_stack.push(EvalOp::ShortCircuitOr { 
262                            right_expr: right, 
263                            ctx_id 
264                        });
265                        self.op_stack.push(EvalOp::Eval { expr: left, ctx_id });
266                    }
267                }
268            }
269            
270            AstExpr::Conditional { condition, true_branch, false_branch } => {
271                self.op_stack.push(EvalOp::TernaryCondition { 
272                    true_branch,
273                    false_branch,
274                    ctx_id,
275                });
276                self.op_stack.push(EvalOp::Eval { expr: condition, ctx_id });
277            }
278            
279            AstExpr::Function { name, args } => {
280                // Special handling for short-circuit operators
281                match (*name, args.len()) {
282                    ("&&", 2) => {
283                        // Short-circuit AND: evaluate left first, then right only if left is true
284                        self.op_stack.push(EvalOp::ShortCircuitAnd { 
285                            right_expr: &args[1], 
286                            ctx_id 
287                        });
288                        self.op_stack.push(EvalOp::Eval { expr: &args[0], ctx_id });
289                    }
290                    ("||", 2) => {
291                        // Short-circuit OR: evaluate left first, then right only if left is false
292                        self.op_stack.push(EvalOp::ShortCircuitOr { 
293                            right_expr: &args[1], 
294                            ctx_id 
295                        });
296                        self.op_stack.push(EvalOp::Eval { expr: &args[0], ctx_id });
297                    }
298                    _ => {
299                        // All other function calls go through the same path to support overrides
300                        // The parser represents operators like ^, +, -, etc. as function calls
301                        // So we treat them all uniformly to allow user overrides
302                        let fname = name.try_into_function_name()?;
303                        
304                        if args.is_empty() {
305                            // No arguments to evaluate
306                            self.op_stack.push(EvalOp::ApplyFunction {
307                                name: fname,
308                                args_needed: 0,
309                                args_collected: Vec::new(),
310                                ctx_id,
311                            });
312                        } else {
313                            // Push collection operation
314                            self.op_stack.push(EvalOp::CollectFunctionArgs {
315                                name: fname,
316                                total_args: args.len(),
317                                args_so_far: Vec::new(),
318                                ctx_id,
319                            });
320                            
321                            // Push argument evaluations in reverse order
322                            for arg in args.iter().rev() {
323                                self.op_stack.push(EvalOp::Eval { expr: arg, ctx_id });
324                            }
325                        }
326                    }
327                }
328            }
329            
330            AstExpr::Array { name, index } => {
331                let array_name = name.try_into_heapless()?;
332                
333                self.op_stack.push(EvalOp::AccessArray { array_name, ctx_id });
334                self.op_stack.push(EvalOp::Eval { expr: index, ctx_id });
335            }
336            
337            AstExpr::Attribute { base, attr } => {
338                let obj_name = base.try_into_heapless()?;
339                let attr_name = attr.try_into_heapless()?;
340                
341                self.op_stack.push(EvalOp::AccessAttribute { 
342                    object_name: obj_name,
343                    attr_name,
344                    ctx_id,
345                });
346            }
347        }
348        
349        Ok(())
350    }

352    /// Process variable lookup
353    fn process_variable_lookup(&mut self, name: HString, ctx_id: usize) -> Result<(), ExprError> {
354        // Check parameter overrides first (highest priority for batch evaluation)
355        if let Some(ref overrides) = self.param_overrides {
356            if let Some(&value) = overrides.get(&name) {
357                self.value_stack.push(value);
358                return Ok(());
359            }
360        }
361        
362        // Try context stack next
363        if let Some(value) = self.ctx_stack.lookup_variable(ctx_id, &name) {
364            self.value_stack.push(value);
365            return Ok(());
366        }
367        
368        // Check if it's a built-in constant
369        let value = match name.as_str() {
370            "pi" | "PI" => core::f64::consts::PI as Real,
371            "e" | "E" => core::f64::consts::E as Real,
372            "tau" | "TAU" => 2.0 * core::f64::consts::PI as Real,
373            _ => {
374                // Check if this looks like a function name
375                let is_potential_function_name = match name.as_str() {
376                    "sin" | "cos" | "tan" | "asin" | "acos" | "atan" | "atan2" | 
377                    "sinh" | "cosh" | "tanh" | "exp" | "log" | "log10" | "ln" | 
378                    "sqrt" | "abs" | "ceil" | "floor" | "pow" | "neg" | "," | 
379                    "comma" | "+" | "-" | "*" | "/" | "%" | "^" | "max" | "min" |
380                    "<" | ">" | "<=" | ">=" | "==" | "!=" => true,
381                    _ => false,
382                };
383                
384                if is_potential_function_name && name.len() > 1 {
385                    return Err(ExprError::Syntax(format!(
386                        "Function '{}' used without arguments",
387                        name
388                    )));
389                }
390                
391                return Err(ExprError::UnknownVariable { name: name.to_string() })
392            }
393        };
394        
395        self.value_stack.push(value);
396        Ok(())
397    }

399    /// Process array access
400    fn process_array_access(&mut self, array_name: HString, index: Real, ctx_id: usize) -> Result<(), ExprError> {
401        let idx = index as usize;
402        
403        if let Some(ctx) = self.ctx_stack.get_context(ctx_id) {
404            if let Some(array) = ctx.arrays.get(&array_name) {
405                if idx < array.len() {
406                    self.value_stack.push(array[idx]);
407                    return Ok(());
408                } else {
409                    return Err(ExprError::ArrayIndexOutOfBounds { 
410                        name: array_name.to_string(), 
411                        index: idx,
412                        len: array.len()
413                    });
414                }
415            }
416        }
417        
418        Err(ExprError::UnknownVariable { 
419            name: array_name.to_string() 
420        })
421    }

423    /// Process attribute access
424    fn process_attribute_access(
425        &mut self, 
426        object_name: HString, 
427        attr_name: HString, 
428        ctx_id: usize
429    ) -> Result<(), ExprError> {
430        if let Some(ctx) = self.ctx_stack.get_context(ctx_id) {
431            if let Some(obj_attrs) = ctx.attributes.get(&object_name) {
432                if let Some(&value) = obj_attrs.get(&attr_name) {
433                    self.value_stack.push(value);
434                    return Ok(());
435                }
436            }
437        }
438        
439        Err(ExprError::AttributeNotFound { 
440            base: object_name.to_string(),
441            attr: attr_name.to_string(),
442        })
443    }

445    /// Process function call
446    fn process_function_call(
447        &mut self,
448        name: FunctionName,
449        args_needed: usize,
450        args: Vec<Real>,
451        ctx_id: usize,
452    ) -> Result<(), ExprError> {
453        // Get context
454        let ctx = self.ctx_stack.get_context(ctx_id)
455            .ok_or_else(|| ExprError::Other("Invalid context ID".to_string()))?;
456        
457        // Try expression function first (highest priority) - clone it to avoid borrowing issues
458        let expr_func = ctx.get_expression_function(&name).cloned();
459        if let Some(func) = expr_func {
460            if args.len() != func.params.len() {
461                return Err(ExprError::InvalidFunctionCall {
462                    name: name.to_string(),
463                    expected: func.params.len(),
464                    found: args.len(),
465                });
466            }
467            
468            // Create new context for function evaluation
469            let mut func_ctx = EvalContext::new();
470            
471            // Set parameters
472            for (param, value) in func.params.iter().zip(args.iter()) {
473                func_ctx.set_parameter(param, *value)?;
474            }
475            
476            // Copy function registry
477            func_ctx.function_registry = ctx.function_registry.clone();
478            
479            // Parse expression function on-demand if we have an arena
480            if let Some(arena) = self.arena {
481                // Check if we've already parsed this function
482                let func_key = func.name.try_into_heapless()?;
483                
484                let ast = if let Some(&cached_ast) = self.expr_func_cache.get(&func_key) {
485                    cached_ast
486                } else {
487                    // Parse the expression function body into the arena
488                    let param_names: Vec<String> = func.params.clone();
489                    let parsed_ast = crate::engine::parse_expression_arena_with_reserved(
490                        &func.expression,
491                        arena,
492                        Some(&param_names)
493                    )?;
494                    
495                    // Allocate the AST in the arena
496                    let arena_ast = arena.alloc(parsed_ast);
497                    self.expr_func_cache.insert(func_key.clone(), arena_ast);
498                    // Return the reference - no move needed
499                    &*arena_ast
500                };
501                
502                // Push the function's AST for evaluation with the new context
503                let func_ctx_id = self.ctx_stack.push_context(Some(Rc::new(func_ctx)))?;
504                self.op_stack.push(EvalOp::Eval { 
505                    expr: ast, 
506                    ctx_id: func_ctx_id 
507                });
508                return Ok(());
509            } else {
510                // No arena available for expression functions
511                return Err(ExprError::Other(
512                    "Expression functions require an arena-enabled evaluator".to_string()
513                ));
514            }
515        }
516        
517        // Try native function second
518        if let Some(func) = ctx.get_native_function(&name) {
519            if args.len() != func.arity {
520                return Err(ExprError::InvalidFunctionCall {
521                    name: name.to_string(),
522                    expected: func.arity,
523                    found: args.len(),
524                });
525            }
526            
527            let owned_fn = OwnedNativeFunction::from(func);
528            let result = (owned_fn.implementation)(&args);
529            self.value_stack.push(result);
530            return Ok(());
531        }
532        
533        // Try built-in functions
534        if let Some(result) = self.try_builtin_function(&name, &args)? {
535            self.value_stack.push(result);
536            return Ok(());
537        }
538        
539        Err(ExprError::UnknownFunction { name: name.to_string() })
540    }

542    /// Try to evaluate built-in functions
543    fn try_builtin_function(&self, name: &FunctionName, args: &[Real]) -> Result<Option<Real>, ExprError> {
544        let result = match (name.as_str(), args.len()) {
545            // Unary operators
546            ("-", 1) => Some(-args[0]),
547            ("!", 1) => Some(if args[0] == 0.0 { 1.0 } else { 0.0 }),
548            ("neg", 1) => Some(-args[0]),
549            
550            // Binary operators (treated as functions for override support)
551            ("+", 2) => Some(args[0] + args[1]),
552            ("-", 2) => Some(args[0] - args[1]),
553            ("*", 2) => Some(args[0] * args[1]),
554            ("/", 2) => {
555                if args[1] == 0.0 {
556                    return Err(ExprError::DivideByZero);
557                }
558                Some(args[0] / args[1])
559            },
560            ("%", 2) => {
561                if args[1] == 0.0 {
562                    return Err(ExprError::DivideByZero);
563                }
564                Some(args[0] % args[1])
565            },
566            ("^", 2) | ("**", 2) | ("pow", 2) => {
567                #[cfg(feature = "libm")]
568                {
569                    Some(crate::functions::pow(args[0], args[1]))
570                }
571                #[cfg(not(feature = "libm"))]
572                {
573                    None // Will be handled by registered functions
574                }
575            },
576            
577            // Comparison operators
578            ("<", 2) => Some(if args[0] < args[1] { 1.0 } else { 0.0 }),
579            (">", 2) => Some(if args[0] > args[1] { 1.0 } else { 0.0 }),
580            ("<=", 2) => Some(if args[0] <= args[1] { 1.0 } else { 0.0 }),
581            (">=", 2) => Some(if args[0] >= args[1] { 1.0 } else { 0.0 }),
582            ("==", 2) => Some(if args[0] == args[1] { 1.0 } else { 0.0 }),
583            ("!=", 2) => Some(if args[0] != args[1] { 1.0 } else { 0.0 }),
584            
585            // Math functions - only available with libm or std
586            #[cfg(feature = "libm")]
587            ("sin", 1) => Some(crate::functions::sin(args[0], 0.0)),
588            #[cfg(feature = "libm")]
589            ("cos", 1) => Some(crate::functions::cos(args[0], 0.0)),
590            #[cfg(feature = "libm")]
591            ("tan", 1) => Some(crate::functions::tan(args[0], 0.0)),
592            #[cfg(feature = "libm")]
593            ("asin", 1) => Some(crate::functions::asin(args[0], 0.0)),
594            #[cfg(feature = "libm")]
595            ("acos", 1) => Some(crate::functions::acos(args[0], 0.0)),
596            #[cfg(feature = "libm")]
597            ("atan", 1) => Some(crate::functions::atan(args[0], 0.0)),
598            #[cfg(feature = "libm")]
599            ("atan2", 2) => Some(crate::functions::atan2(args[0], args[1])),
600            #[cfg(feature = "libm")]
601            ("sinh", 1) => Some(crate::functions::sinh(args[0], 0.0)),
602            #[cfg(feature = "libm")]
603            ("cosh", 1) => Some(crate::functions::cosh(args[0], 0.0)),
604            #[cfg(feature = "libm")]
605            ("tanh", 1) => Some(crate::functions::tanh(args[0], 0.0)),
606            #[cfg(feature = "libm")]
607            ("exp", 1) => Some(crate::functions::exp(args[0], 0.0)),
608            #[cfg(feature = "libm")]
609            ("ln", 1) | ("log", 1) => Some(crate::functions::ln(args[0], 0.0)),
610            #[cfg(feature = "libm")]
611            ("log10", 1) => Some(crate::functions::log10(args[0], 0.0)),
612            #[cfg(feature = "libm")]
613            ("log2", 1) => Some(libm::log2(args[0])),
614            #[cfg(feature = "libm")]
615            ("sqrt", 1) => Some(crate::functions::sqrt(args[0], 0.0)),
616            ("abs", 1) => Some(crate::functions::abs(args[0], 0.0)),
617            #[cfg(feature = "libm")]
618            ("floor", 1) => Some(crate::functions::floor(args[0], 0.0)),
619            #[cfg(feature = "libm")]
620            ("ceil", 1) => Some(crate::functions::ceil(args[0], 0.0)),
621            #[cfg(feature = "libm")]
622            ("round", 1) => Some(libm::round(args[0])),
623            #[cfg(feature = "libm")]
624            ("trunc", 1) => Some(libm::trunc(args[0])),
625            ("sign", 1) => Some(if args[0] > 0.0 { 1.0 } else if args[0] < 0.0 { -1.0 } else { 0.0 }),
626            
627            // Multi-argument functions
628            ("min", 2) => Some(args[0].min(args[1])),
629            ("max", 2) => Some(args[0].max(args[1])),
630            #[cfg(feature = "libm")]
631            ("hypot", 2) => Some(libm::hypot(args[0], args[1])),
632            ("fmod", 2) => Some(args[0] % args[1]),
633            (",", 2) | ("comma", 2) => Some(args[1]), // Comma operator returns the second value
634            
635            // Not a built-in
636            _ => None,
637        };
638        
639        Ok(result)
640    }
641    
642    /// Pop a value from the value stack
643    fn pop_value(&mut self) -> Result<Real, ExprError> {
644        self.value_stack.pop()
645            .ok_or_else(|| ExprError::Other("Value stack underflow".to_string()))
646    }
647    
648    /// Set parameter overrides for batch evaluation.
649    /// These take precedence over context variables during lookup.
650    pub fn set_param_overrides(&mut self, params: FnvIndexMap<HString, Real, 16>) {
651        self.param_overrides = Some(params);
652    }
653    
654    /// Clear parameter overrides.
655    pub fn clear_param_overrides(&mut self) {
656        self.param_overrides = None;
657    }
658    
659    /// Execute a function with parameter overrides, ensuring they are cleared afterwards.
660    /// This provides RAII-style cleanup for safe batch evaluation.
661    pub fn with_param_overrides<F, R>(&mut self, params: FnvIndexMap<HString, Real, 16>, f: F) -> R 
662    where 
663        F: FnOnce(&mut Self) -> R 
664    {
665        let old_overrides = self.param_overrides.take();
666        self.param_overrides = Some(params);
667        let result = f(self);
668        self.param_overrides = old_overrides;
669        result
670    }
671}

673/// Evaluate an expression using a provided engine (avoids engine allocation).
674///
675/// This function is useful for batch evaluation where the same engine can be
676/// reused across multiple evaluations, avoiding the overhead of creating a
677/// new engine for each evaluation.
678///
679/// # Parameters
680///
681/// * `ast` - The abstract syntax tree to evaluate
682/// * `ctx` - Optional evaluation context containing variables and functions
683/// * `engine` - Mutable reference to an existing evaluation engine
684///
685/// # Returns
686///
687/// The result of evaluating the expression, or an error if evaluation fails
688///
689/// # Example
690///
691/// ```
692/// use exp_rs::eval::{eval_with_engine, EvalEngine};
693/// use exp_rs::engine::parse_expression;
694/// use exp_rs::context::EvalContext;
695/// use std::rc::Rc;
696///
697/// let ast = parse_expression("2 + 3").unwrap();
698/// let mut engine = EvalEngine::new();
699/// let result = eval_with_engine(&ast, None, &mut engine).unwrap();
700/// assert_eq!(result, 5.0);
701/// ```
702pub fn eval_with_engine<'arena>(
703    ast: &'arena AstExpr<'arena>,
704    ctx: Option<Rc<EvalContext>>,
705    engine: &mut EvalEngine<'arena>,
706) -> Result<Real, ExprError> {
707    engine.eval(ast, ctx)
708}
