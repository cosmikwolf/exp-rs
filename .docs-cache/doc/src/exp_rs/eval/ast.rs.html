21pub fn eval_ast<'arena>(ast: &'arena AstExpr<'arena>, ctx: Option<Rc<EvalContext>>) -> Result<Real, ExprError> {
22    // Use the iterative evaluator - this eliminates stack overflow issues
23    // and provides better performance than the recursive approach
24    crate::eval::iterative::eval_iterative(ast, ctx)
25}

27pub fn eval_ast_inner<'arena>(
28    ast: &'arena AstExpr<'arena>,
29    ctx: Option<Rc<EvalContext>>,
30    func_cache: &mut BTreeMap<String, Option<FunctionCacheEntry>>,
31    var_cache: &mut BTreeMap<String, Real>,
32) -> Result<Real, ExprError> {
33    // We'll be more selective about when to increment the recursion counter
34    // Only increment for function calls, logical operations, and conditionals, as these are the ones that can cause stack overflow.
35    //
36    // LogicalOp and Conditional nodes are tracked because:
37    // 1. They may contain recursive function calls in their operands/branches
38    // 2. Short-circuit evaluation can bypass potential infinite recursion in the non-executed branches
39    // 3. Complex logical or conditional expressions might cause stack overflows even with short-circuit behavior
40    let should_track = matches!(ast, AstExpr::Function { .. } | AstExpr::LogicalOp { .. } | AstExpr::Conditional { .. });
41
42    // Check and increment recursion depth only for function calls and logical operations.
43    // This prevents stack overflows from infinite recursion while still allowing
44    // legitimate use of recursive functions and complex logical expressions.
45    if should_track {
46        check_and_increment_recursion_depth()?;
47    }
48
49    // Store result to ensure we always decrement the counter if needed
50    let result = match ast {
51        AstExpr::Constant(val) => Ok(*val),
52        AstExpr::Variable(name) => eval_variable(name, ctx.clone(), var_cache),
53        AstExpr::Function { name, args } => {
54            eval_function(name, args, ctx.clone(), func_cache, var_cache)
55        }
56        AstExpr::Array { name, index } => {
57            eval_array(name, index, ctx.clone(), func_cache, var_cache)
58        }
59        AstExpr::Attribute { base, attr } => eval_attribute(base, attr, ctx),
60        AstExpr::LogicalOp { op, left, right } => {
61            // Implement short-circuit evaluation for logical operators
62            // Short-circuit evaluation provides two key benefits:
63            // 1. Performance - avoid unnecessary computation
64            // 2. Error prevention - potentially skip evaluating expressions that would cause errors
65            match op {
66                crate::types::LogicalOperator::And => {
67                    // Evaluate left side first
68                    let left_val = eval_ast_inner(left, ctx.clone(), func_cache, var_cache)?;
69
70                    // Short-circuit if left is false (0.0)
71                    // This is important for both performance and error prevention:
72                    // - If left is false, the overall result must be false regardless of right
73                    // - Any potential errors in the right operand are completely avoided
74                    if left_val == 0.0 {
75                        Ok(0.0)
76                    } else {
77                        // Only evaluate right side if left is true (non-zero)
78                        // Note: Errors in the right side are still propagated upward
79                        let right_val = eval_ast_inner(right, ctx, func_cache, var_cache)?;
80                        // Result is true (1.0) only if both are true (non-zero)
81                        // We normalize the result to 1.0 for consistency
82                        Ok(if right_val != 0.0 { 1.0 } else { 0.0 })
83                    }
84                }
85                crate::types::LogicalOperator::Or => {
86                    // Evaluate left side first
87                    let left_val = eval_ast_inner(left, ctx.clone(), func_cache, var_cache)?;
88
89                    // Short-circuit if left is true (non-zero)
90                    // This is important for both performance and error prevention:
91                    // - If left is true, the overall result must be true regardless of right
92                    // - Any potential errors in the right operand are completely avoided
93                    if left_val != 0.0 {
94                        Ok(1.0)
95                    } else {
96                        // Only evaluate right side if left is false (zero)
97                        // Note: Errors in the right side are still propagated upward
98                        let right_val = eval_ast_inner(right, ctx, func_cache, var_cache)?;
99                        // Result is true (1.0) if either is true (non-zero)
100                        // We normalize the result to 1.0 for consistency
101                        Ok(if right_val != 0.0 { 1.0 } else { 0.0 })
102                    }
103                }
104            }
105        },
106        AstExpr::Conditional { condition, true_branch, false_branch } => {
107            // Evaluate the condition first
108            let condition_val = eval_ast_inner(condition, ctx.clone(), func_cache, var_cache)?;
109            
110            // Short-circuit to the appropriate branch based on the condition
111            // This provides both performance optimization and error prevention:
112            // - Only one branch is evaluated, avoiding unnecessary computation
113            // - Potential errors in the non-executed branch are completely avoided
114            if condition_val != 0.0 {
115                // Condition is true (non-zero), evaluate the true branch
116                eval_ast_inner(true_branch, ctx, func_cache, var_cache)
117            } else {
118                // Condition is false (zero), evaluate the false branch
119                eval_ast_inner(false_branch, ctx, func_cache, var_cache)
120            }
121        }
122    };
123
124    // Only decrement if we incremented
125    if should_track {
126        decrement_recursion_depth();
127    }
128
129    result
130}
