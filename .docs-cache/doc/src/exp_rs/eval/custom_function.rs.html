48    name: &str,
49    args: &[AstExpr],
50    ctx: Option<Rc<EvalContext>>,
51    func_cache: &mut BTreeMap<String, Option<FunctionCacheEntry>>,
52    var_cache: &mut BTreeMap<String, Real>,
53    func: &F,
54) -> Result<Real, ExprError>
55where
56    F: CustomFunction,
57{
58    if args.len() != func.params().len() {
59        return Err(ExprError::InvalidFunctionCall {
60            name: name.to_string(),
61            expected: func.params().len(),
62            found: args.len(),
63        });
64    }
65    // First evaluate all arguments in the parent context
66    let mut arg_values = Vec::with_capacity(args.len());
67    for arg in args {
68        let arg_val = eval_ast_inner(arg, ctx.clone(), func_cache, var_cache)?;
69        arg_values.push(arg_val);
70    }
71
72    // Create a fresh context with the function arguments
73    let mut func_ctx = EvalContext::new();
74
75    // Add all parameters with their values
76    for (i, param_name) in func.params().iter().enumerate() {
77        func_ctx.set_parameter(param_name, arg_values[i]);
78    }
79
80    // Copy function registry from parent to access all functions
81    if let Some(parent) = &ctx {
82        func_ctx.function_registry = parent.function_registry.clone();
83    }
84
85    // We'll use the passed-in function cache directly
86
87    // Parse body string on demand
88    // TODO: This needs to be updated to use arena allocation
89    let param_names_str: Vec<String> = func.params().iter().map(|c| c.to_string()).collect();
90    panic!("Expression functions need to be converted to template pattern - cannot parse without arena");

94    // We need to create a special version of eval_ast_inner that knows how to handle
95    // variables in this custom function scope
96    fn eval_custom_function_ast<'b>(
97        ast: &AstExpr,
98        func_ctx: &EvalContext,
99        global_ctx: Option<&Rc<EvalContext>>,
100        func_cache: &mut BTreeMap<String, Option<FunctionCacheEntry>>,
101        var_cache: &mut BTreeMap<String, Real>,
102    ) -> Result<Real, ExprError> {
103        // Track recursion depth for function calls and logical operations
104        // Track recursion depth for function calls, logical operations, and conditionals.
105        // We include LogicalOp and Conditional in recursion tracking for two main reasons:
106        // 1. They can contain recursive function calls in operands/branches
107        // 2. Short-circuit evaluation might involve complex logic that could lead to stack overflows
108        let should_track = matches!(ast, AstExpr::Function { .. } | AstExpr::LogicalOp { .. } | AstExpr::Conditional { .. });
109
110        // Check and increment recursion depth only if needed.
111        // This protects against infinite recursion in expression functions and
112        // complex logical operations that might cause stack overflows.
113        if should_track {
114            check_and_increment_recursion_depth()?;
115        }
116
117        // Store result to ensure we always decrement the counter if needed
118        let result = match ast {
119            AstExpr::Constant(val) => Ok(*val),
120            AstExpr::Conditional { condition, true_branch, false_branch } => {
121                // Evaluate the condition first
122                let condition_val = eval_custom_function_ast(condition, func_ctx, global_ctx, func_cache, var_cache)?;
123                
124                // Short-circuit to the appropriate branch based on the condition
125                if condition_val != 0.0 {
126                    // Condition is true (non-zero), evaluate the true branch
127                    eval_custom_function_ast(true_branch, func_ctx, global_ctx, func_cache, var_cache)
128                } else {
129                    // Condition is false (zero), evaluate the false branch
130                    eval_custom_function_ast(false_branch, func_ctx, global_ctx, func_cache, var_cache)
131                }
132            },
133            AstExpr::Variable(name) => {
134                // First check if the variable is a parameter in the function context
135                if let Ok(key) = name.clone().try_into_heapless() {
136                    if let Some(val) = func_ctx.variables.get(&key) {
137                        return Ok(*val);
138                    }
139                }
140                // If not found, delegate to normal variable lookup in global context
141                if let Some(ctx) = global_ctx {
142                    eval_variable(name, Some(ctx.clone()), var_cache)
143                } else {
144                    // No contexts available
145                    Err(ExprError::UnknownVariable {
146                        name: name.to_string(),
147                    })
148                }
149            }
150            AstExpr::Function { name, args } => {
151                // Create a vector to store evaluated arguments
152                let mut arg_values = Vec::with_capacity(args.len());
153
154                // Evaluate each argument in the proper context
155                for arg in args.iter() {
156                    // Recursively call our custom evaluator
157                    let arg_val =
158                        eval_custom_function_ast(arg, func_ctx, global_ctx, func_cache, var_cache)?;
159                    arg_values.push(arg_val);
160                }
161
162                // Once we have all argument values, find and evaluate the function
163                if let Some(native_fn) = func_ctx.get_native_function(name) {
164                    // Native function
165                    if arg_values.len() != native_fn.arity {
166                        Err(ExprError::InvalidFunctionCall {
167                            name: name.to_string(),
168                            expected: native_fn.arity,
169                            found: arg_values.len(),
170                        })
171                    } else {
172                        // Convert to OwnedNativeFunction
173                        let owned_fn = OwnedNativeFunction::from(native_fn);
174                        Ok((owned_fn.implementation)(&arg_values))
175                    }
176                } else if let Some(expr_fn) = func_ctx.get_expression_function(name) {
177                    // Expression function (recursive case)
178                    // This is the critical case for tracking recursion depth
179                    if arg_values.len() != expr_fn.params.len() {
180                        Err(ExprError::InvalidFunctionCall {
181                            name: name.to_string(),
182                            expected: expr_fn.params.len(),
183                            found: arg_values.len(),
184                        })
185                    } else {
186                        // Explicitly check recursion depth before proceeding to nested evaluation
187                        // This provides an additional safety check before entering a potentially
188                        // recursive function evaluation path
189                        check_and_increment_recursion_depth()?;
190                        
191                        // Create a fresh context for this function call
192                        let mut nested_func_ctx = EvalContext::new();
193
194                        // Add parameters
195                        for (i, param_name) in expr_fn.params.iter().enumerate() {
196                            nested_func_ctx.set_parameter(param_name, arg_values[i]);
197                        }
198
199                        // Copy function registry
200                        if let Some(parent) = global_ctx {
201                            nested_func_ctx.function_registry = parent.function_registry.clone();
202                        }
203
204                        // TODO: Expression functions need arena conversion
205                        panic!("Expression functions need to be converted to template pattern")
206                    }
207                } else {
208                    // Not found in the contexts, try built-in functions
209                    // Just delegate to eval_function in the global context
210                    let ast_args: Vec<AstExpr> = args
211                        .iter()
212                        .zip(arg_values.iter())
213                        .map(|(_, val)| AstExpr::Constant(*val))
214                        .collect();
215
216                    eval_function(name, &ast_args, global_ctx.cloned(), func_cache, var_cache)
217                }
218            }
219            AstExpr::Array { name, index } => {
220                let idx_val =
221                    eval_custom_function_ast(index, func_ctx, global_ctx, func_cache, var_cache)?
222                        as usize;
223
224                // First check function context
225                if let Some(arr) = func_ctx.get_array(name) {
226                    if idx_val < arr.len() {
227                        Ok(arr[idx_val])
228                    } else {
229                        Err(ExprError::ArrayIndexOutOfBounds {
230                            name: name.to_string(),
231                            index: idx_val,
232                            len: arr.len(),
233                        })
234                    }
235                } else if let Some(global) = global_ctx {
236                    // Then check global context
237                    if let Some(arr) = global.get_array(name) {
238                        if idx_val < arr.len() {
239                            Ok(arr[idx_val])
240                        } else {
241                            Err(ExprError::ArrayIndexOutOfBounds {
242                                name: name.to_string(),
243                                index: idx_val,
244                                len: arr.len(),
245                            })
246                        }
247                    } else {
248                        Err(ExprError::UnknownVariable {
249                            name: name.to_string(),
250                        })
251                    }
252                } else {
253                    Err(ExprError::UnknownVariable {
254                        name: name.to_string(),
255                    })
256                }
257            }
258            AstExpr::Attribute { base, attr } => {
259                // Delegate to eval_attribute
260                if let Some(global) = global_ctx {
261                    eval_attribute(base, attr, Some(global.clone()))
262                } else {
263                    Err(ExprError::AttributeNotFound {
264                        base: base.to_string(),
265                        attr: attr.to_string(),
266                    })
267                }
268            }
269            AstExpr::LogicalOp { op, left, right } => {
270                // Implement short-circuit evaluation for logical operators in custom function context
271                // This is particularly important when logical operators are used inside user-defined functions,
272                // as they may involve recursive calls that need to be properly tracked
273                match op {
274                    crate::types::LogicalOperator::And => {
275                        // Evaluate left side first
276                        let left_val = eval_custom_function_ast(
277                            left, func_ctx, global_ctx, func_cache, var_cache,
278                        )?;
279
280                        // Short-circuit if left is false (0.0)
281                        // This helps prevent unnecessary recursion and potential stack overflow
282                        if left_val == 0.0 {
283                            Ok(0.0)
284                        } else {
285                            // Only evaluate right side if left is true (non-zero)
286                            let right_val = eval_custom_function_ast(
287                                right, func_ctx, global_ctx, func_cache, var_cache,
288                            )?;
289                            // Result is true (1.0) only if both are true (non-zero)
290                            // We normalize the result to 1.0 for consistency
291                            Ok(if right_val != 0.0 { 1.0 } else { 0.0 })
292                        }
293                    }
294                    crate::types::LogicalOperator::Or => {
295                        // Evaluate left side first
296                        let left_val = eval_custom_function_ast(
297                            left, func_ctx, global_ctx, func_cache, var_cache,
298                        )?;
299
300                        // Short-circuit if left is true (non-zero)
301                        // This helps prevent unnecessary recursion and potential stack overflow
302                        if left_val != 0.0 {
303                            Ok(1.0)
304                        } else {
305                            // Only evaluate right side if left is false (zero)
306                            let right_val = eval_custom_function_ast(
307                                right, func_ctx, global_ctx, func_cache, var_cache,
308                            )?;
309                            // Result is true (1.0) if either is true (non-zero)
310                            // We normalize the result to 1.0 for consistency
311                            Ok(if right_val != 0.0 { 1.0 } else { 0.0 })
312                        }
313                    }
314                }
315            }
316        };
317
318        // Only decrement if we incremented
319        if should_track {
320            decrement_recursion_depth();
321        }
322
323        result
324    }
